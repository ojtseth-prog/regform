require("dotenv").config();
const express = require("express");
const mysql = require("mysql2");
const cors = require("cors");

const app = express();
const port = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(express.json());

const pool = mysql.createPool({
  host: process.env.DB_HOST || "corerxinfo.impactprotech.host",
  user: process.env.DB_USER || "u826896242_registration",
  password: process.env.DB_PASSWORD || "IMPACTPROTECH@2023",
  database: process.env.DB_NAME || "registration",
  port: process.env.DB_PORT || 3306,
  waitForConnections: true,
  connectionLimit: 10
});

const db = pool.promise();

// Test database connection
async function testConnection() {
  try {
    const connection = await db.getConnection();
    console.log("✅ Database connected successfully");
    connection.release();
  } catch (err) {
    console.error("❌ Database connection failed:", err.message);
    console.log("Please check your database credentials and host");
  }
}

testConnection();

app.post("/api/register", async (req, res) => {
  const { pharmacy_name, address, contact_number, email_address } = req.body;

  if (!pharmacy_name || !address || !contact_number || !email_address) {
    return res.status(400).json({ error: "All pharmacy fields are required" });
  }

  try {
    const query = "INSERT INTO registration (pharmacy_name, address, contact_number, email_address) VALUES (?, ?, ?, ?)";
    const [result] = await db.query(query, [pharmacy_name, address, contact_number, email_address]);
    res.status(201).json({ message: "Registration successful", id: result.insertId });
  } catch (err) {
    console.error("Error inserting data:", err);
    res.status(500).json({ error: "Database error: " + err.message });
  }
});

// GET Endpoint for checking data
app.get("/api/data", async (req, res) => {
  try {
    const [results] = await db.query("SELECT * FROM registration");
    res.json(results);
  } catch (err) {
    console.error("Error fetching data:", err);
    res.status(500).json({ error: "Database error" });
  }
});

app.listen(port, () => {
  console.log(`Server running on port ${port}`);
});
